```python
class SearchResult:
```

    ```python
    def extract_snippet(body: str, query: str, *, max_line_len: int = 120) -> str:
        """First line matching `query` (case-insensitive) plus up to one line
        above and one below, joined with '\\n'. Each line truncated to
        max_line_len chars with '...' suffix if exceeded.

        Returns '' when no match.

        Literal or regex? Uses case-insensitive substring match in body
        lines — callers wanting regex snippet highlighting can build their
        own later; v1 keeps snippet extraction safe and simple."""
        if not query or not body:
            return ""
        q = query.lower()
        lines = body.split("\n")
        for i, line in enumerate(lines):
            if q in line.lower():
                start = max(0, i - 1)
                end = min(len(lines), i + 2)
                window = lines[start:end]
                truncated = []
                for ln in window:
                    if len(ln) > max_line_len:
                        truncated.append(ln[:max_line_len - 3] + "...")
                    else:
                        truncated.append(ln)
                return "\n".join(truncated)
        return ""
    ```

    ```python
    def scan_file_with_timeout(
        path: Path,
        query: str,
        *,
        timeout: float = 0.1,
        is_literal: bool,
    ) -> tuple[int, bool]:
        """Return (match_count, timed_out). On ReDoS timeout or file error,
        logs warning to stderr and returns (0, True) or (0, False) respectively.

        Literal path: read file text, lower it, str.count(query.lower()).
        No ReDoS possible — no watchdog needed.

        Regex path: spawn a daemon thread that runs re.findall; parent thread
        waits `timeout` seconds via thread.join(timeout). If thread still alive
        after timeout, we cannot kill it cleanly (Python has no pthread_cancel)
        so we set a flag, log the offending pattern+path, and return (0, True).
        The zombie thread eventually finishes on its own (it is daemon=True so
        it cannot block interpreter shutdown)."""
    ```

    ```python
    def scan_file_with_timeout(path, query, *, timeout=0.1, is_literal):
        try:
            text = path.read_text(encoding="utf-8", errors="replace")
        except OSError as e:
            _log.warning("search: cannot read %s: %s", path, e)
            return (0, False)

        if is_literal:
            return (text.lower().count(query.lower()), False)

        # Regex path with timeout watchdog
        result: dict = {"count": 0, "done": False}
        def _worker():
            try:
                pattern = re.compile(query, re.IGNORECASE)
                result["count"] = len(pattern.findall(text))
            except re.error as e:
                _log.warning("search: invalid regex %r: %s", query, e)
            finally:
                result["done"] = True

        t = threading.Thread(target=_worker, daemon=True)
        t.start()
        t.join(timeout)
        if not result["done"]:
            _log.warning(
                "search: regex %r timed out on %s after %.3fs (possible ReDoS)",
                query, path, timeout,
            )
            return (0, True)
        return (result["count"], False)
    ```

    ```python
    from cyrus.paths import CATEGORIES, category_dir, cyrus_home
    from cyrus.storage import parse_frontmatter

    _log = get_logger(__name__)

    _REDOS_TIMEOUT_SECONDS = 0.1


    @dataclass
    class SearchResult:
        path: Path
        score: float
        snippet: str
        metadata: dict[str, Any] = field(default_factory=dict)


    def search(
        query: str,
        category: str | None = None,
        limit: int = 10,
        since: datetime | None = None,
        tags: list[str] | None = None,
    ) -> list[SearchResult]:
        if not query:
            return []
        if category is not None and category not in CATEGORIES:
            raise ValueError(
                f"Unknown category {category!r}. Valid: {CATEGORIES}"
            )

        roots = (
            [category_dir(category)] if category
            else [cyrus_home() / c for c in CATEGORIES]
        )
        literal = is_literal_query(query)
        now = datetime.now(timezone.utc)

        scored: list[tuple[float, str, SearchResult]] = []
        # Tuple shape: (score, updated_iso, result). heapq.nlargest with default
        # ordering compares element-wise: higher score first, then later iso
        # timestamp first (lexicographic on ISO-8601 == chronological).

        for root in roots:
            if not root.exists():
                continue
            for dirpath, _dirnames, filenames in os.walk(root):
                for fname in filenames:
                    if not fname.endswith(".md"):
                        continue
                    path = Path(dirpath) / fname
                    result = _score_file(
                        path=path,
                        query=query,
                        literal=literal,
                        now=now,
                        since=since,
                        tags_filter=tags,
                    )
                    if result is None:
                        continue
                    scored.append(
                        (result.score, result.metadata.get("updated", ""), result)
                    )

        top = heapq.nlargest(limit, scored)
        return [r for (_score, _updated, r) in top]


    def _score_file(
        *,
        path: Path,
        query: str,
        literal: bool,
        now: datetime,
        since: datetime | None,
        tags_filter: list[str] | None,
    ) -> SearchResult | None:
        """Read, filter, score a single file. Returns None if filtered out or
        no match or unreadable."""
        try:
            text = path.read_text(encoding="utf-8", errors="replace")
        except OSError as e:
            _log.warning("search: cannot read %s: %s", path, e)
            return None

        try:
            metadata, body = parse_frontmatter(text)
        except ValueError as e:
            _log.warning("search: bad frontmatter in %s: %s", path, e)
            return None

        # since filter
        if since is not None:
            updated_str = metadata.get("updated", "")
            if not updated_str or updated_str < since.isoformat(timespec="seconds"):
                return None

        # tags filter (AND logic)
        if tags_filter:
            file_tags = metadata.get("tags", [])
            if not isinstance(file_tags, list):
                return None
            if not all(t in file_tags for t in tags_filter):
                return None

        # tf via timeout-guarded scan (on body only, not frontmatter)
        tf, _timed_out = scan_file_with_timeout(
            path, query,
            timeout=_REDOS_TIMEOUT_SECONDS,
            is_literal=literal,
        )
        # scan_file_with_timeout reads the whole file; for snippet we need
        # the body post-frontmatter, but the match count over the whole file
        # is acceptable here (frontmatter matches are rare and minor).
        if tf <= 0:
            return None

        age_days = _age_days(metadata.get("updated", ""), now)
        score = float(tf) * recency_decay(age_days) * filename_bonus(query, path)

        snippet = extract_snippet(body, query)
        return SearchResult(
            path=path, score=score, snippet=snippet, metadata=metadata,
        )


    def _age_days(updated_iso: str, now: datetime) -> float:
        if not updated_iso:
            return 0.0
        try:
            dt = datetime.fromisoformat(updated_iso)
        except ValueError:
            return 0.0
        if dt.tzinfo is None:
            dt = dt.replace(tzinfo=timezone.utc)
        delta = now - dt
        return delta.total_seconds() / 86400.0
    ```
