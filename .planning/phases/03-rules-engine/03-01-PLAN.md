        from __future__ import annotations
        import json
        import re
        from pathlib import Path
        from typing import Any

        from sekha.storage import parse_frontmatter

        _REQUIRED_FIELDS = ("severity", "triggers", "matches", "pattern")
        _VALID_SEVERITIES = ("block", "warn")

        def _anchor_pattern(raw: str, *, anchored: bool) -> str:
            if not anchored:
                return raw
            # Idempotent: don't double-anchor
            prefix = "" if raw.startswith("^") else "^"
            suffix = "" if raw.endswith("$") else "$"
            return f"{prefix}{raw}{suffix}"

        def _flatten_tool_input(tool_input: dict) -> str:
            # Deterministic JSON dump — sort_keys so pattern match is reproducible
            return json.dumps(tool_input, sort_keys=True, default=str)

        def _compile_rule_pattern(raw: str, *, anchored: bool) -> re.Pattern:
            return re.compile(_anchor_pattern(raw, anchored=anchored), re.IGNORECASE)

        def _parse_rule_file(path: Path) -> "Rule":
            # Import Rule lazily to avoid circular import with sekha.rules
            from sekha.rules import Rule

            text = path.read_text(encoding="utf-8")
            meta, body = parse_frontmatter(text)

            missing = [f for f in _REQUIRED_FIELDS if f not in meta]
            if missing:
                raise ValueError(f"missing field: {missing[0]} in {path}")

            severity = meta["severity"]
            if severity not in _VALID_SEVERITIES:
                raise ValueError(
                    f"invalid severity {severity!r} in {path} (must be block|warn)"
                )

            anchored = bool(meta.get("anchored", True))
            raw_pattern = str(meta["pattern"])
            try:
                compiled = _compile_rule_pattern(raw_pattern, anchored=anchored)
            except re.error as exc:
                raise ValueError(
                    f"invalid regex pattern in {path}: {exc}"
                ) from exc

            triggers_raw = meta["triggers"]
            triggers = tuple(triggers_raw) if isinstance(triggers_raw, list) else (triggers_raw,)
            matches_raw = meta["matches"]
            matches = tuple(matches_raw) if isinstance(matches_raw, list) else (matches_raw,)
            priority = int(meta.get("priority", 0))
            message = str(meta.get("message") or body).strip()

            return Rule(
                name=path.stem,
                severity=severity,
                triggers=triggers,
                matches=matches,
                pattern=compiled,
                priority=priority,
                message=message,
                raw_pattern=raw_pattern,
                anchored=anchored,
            )

        def _dir_cache_key(rules_dir: Path) -> tuple[int, float]:
            if not rules_dir.exists():
                return (0, 0.0)
            files = sorted(p for p in rules_dir.glob("*.md") if p.is_file())
            if not files:
                return (0, 0.0)
            max_mtime = max(p.stat().st_mtime for p in files)
            return (len(files), max_mtime)

        - load_rules(rules_dir, hook_event, tool_name) -> list[Rule]

        over an already-loaded rule list.

        RULES-01 through RULES-08 are implemented here. See
        .planning/phases/03-rules-engine/03-CONTEXT.md for the decision log.
        """
        from __future__ import annotations

        import os
        import re
        from dataclasses import dataclass
        from pathlib import Path
        from typing import Any

        from sekha._rulesutil import (
            _compile_rule_pattern,  # noqa: F401 — re-exported for Phase 5 sekha_add_rule validation
            _dir_cache_key,
            _flatten_tool_input,
            _parse_rule_file,
        )
        from sekha.logutil import get_logger
        from sekha.paths import category_dir

        _log = get_logger(__name__)

        _PAUSE_ENV = "SEKHA_PAUSE"

        # Cache: dict[resolved_dir_str, tuple[cache_key, list[Rule_all]]]
        # We cache the full parsed list per directory, then filter per call.
        _CACHE: dict[str, tuple[tuple[int, float], list["Rule"]]] = {}


        @dataclass(frozen=True)
        class Rule:
            name: str
            severity: str               # "block" | "warn"
            triggers: tuple[str, ...]
            matches: tuple[str, ...]
            pattern: re.Pattern
            priority: int
            message: str
            raw_pattern: str
            anchored: bool


        def clear_cache() -> None:
            """Drop all cached parsed-rule lists. Next load_rules re-reads disk."""
            _CACHE.clear()


        def _paused_names() -> set[str]:
            raw = os.environ.get(_PAUSE_ENV, "")
            return {n.strip() for n in raw.split(",") if n.strip()}


        def _load_all(rules_dir: Path) -> list[Rule]:
            """Parse every .md file; cache-aware. Logs invalid rules to stderr."""
            rules_dir = Path(rules_dir).resolve()
            key = str(rules_dir)
            cache_key = _dir_cache_key(rules_dir)
            cached = _CACHE.get(key)
            if cached is not None and cached[0] == cache_key:
                return cached[1]

            rules: list[Rule] = []
            if rules_dir.exists():
                for path in sorted(rules_dir.glob("*.md")):
                    if not path.is_file():
                        continue
                    try:
                        rules.append(_parse_rule_file(path))
                    except (ValueError, OSError) as exc:
                        _log.error("invalid rule %s: %s", path.as_posix(), exc)
                        continue
            _CACHE[key] = (cache_key, rules)
            return rules


        def load_rules(
            rules_dir: Path,
            hook_event: str,
            tool_name: str,
        ) -> list[Rule]:
            """Load rules scoped to (hook_event, tool_name).

            Applies SEKHA_PAUSE suppression. Strict-parses frontmatter — invalid
            rules are skipped with a loud stderr log rather than silently ignored.
            Returns a list sorted by filename (stable, deterministic).
            """
            paused = _paused_names()
            all_rules = _load_all(Path(rules_dir))
            out: list[Rule] = []
            for rule in all_rules:
                if rule.name in paused:
                    continue
                if hook_event not in rule.triggers:
                    continue
                if "*" not in rule.matches and tool_name not in rule.matches:
                    continue
                out.append(rule)
            return out


        def evaluate(rules: list[Rule], tool_input: dict) -> Rule | None:
            """Return the winning rule per precedence, or None.

            Precedence: block > warn, then highest priority, then first (stable
            sorted order — rules come from load_rules sorted by filename). Ties
            log a stderr warning with both rule names.
            """
            if not rules:
                return None
            flat = _flatten_tool_input(tool_input)
            matched = [r for r in rules if r.pattern.search(flat)]
            if not matched:
                return None

            # Sort key: (severity_rank, -priority). severity_rank: block=0, warn=1.
            def rank(r: Rule) -> tuple[int, int]:
                return (0 if r.severity == "block" else 1, -r.priority)

            matched.sort(key=rank)  # stable — preserves load_rules's filename order on ties
            winner = matched[0]
            # Detect ties on same (severity, priority)
            winner_key = (winner.severity, winner.priority)
            tied = [r for r in matched if (r.severity, r.priority) == winner_key]
            if len(tied) > 1:
                others = [r.name for r in tied if r.name != winner.name]
                _log.warning(
                    "sekha.rules: tie between %s and %s, using %s",
                    winner.name,
                    ", ".join(others),
                    winner.name,
                )
            return winner


        def test_rule(rule_name: str, tool_name: str, tool_input: dict) -> dict[str, Any]:
            """Dry-run a single rule by name. Loads from sekha_home()/rules/.

            Returns {"matched": bool, "severity": str, "message": str, "rule": str}.
            Raises FileNotFoundError if the rule file doesn't exist.
            """
            rules_dir = category_dir("rules")
            rule_path = rules_dir / f"{rule_name}.md"
            if not rule_path.exists():
                raise FileNotFoundError(f"rule not found: {rule_path}")
            rule = _parse_rule_file(rule_path)
            flat = _flatten_tool_input(tool_input)
            matched = bool(rule.pattern.search(flat)) and (
                "*" in rule.matches or tool_name in rule.matches
            )
            return {
                "matched": matched,
                "severity": rule.severity,
                "message": rule.message,
                "rule": rule.name,
            }
