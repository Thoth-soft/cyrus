<objective>
Build the `sekha.hook` differentiator — a short-lived Python process invoked by Claude Code's PreToolUse hook — plus the `sekha.cli` argparse router that dispatches to it. This plan lands the entire block/warn/allow decision pipeline, fail-open policy, kill switch, and `sekha hook enable/disable` subcommands. Performance bench and integration-test fixtures ship in plan 04-02.

Purpose: This is the project's moat. If the hook can't emit the correct permissionDecision JSON and survive being invoked 100+ times per Claude Code session without locking the user out, the project has no reason to exist. Fail-open + kill switch are non-negotiable — a bug in Sekha must never stop the user from using their own tools.

       ```python
       """Private helpers for sekha.hook. Lightweight: JSON I/O, fail-open logging, kill-switch bookkeeping."""
       from __future__ import annotations
       import json, sys, traceback
       from datetime import datetime, timezone
       from pathlib import Path
       from sekha.paths import sekha_home

       _KILL_WINDOW_SECONDS = 600  # 10 minutes
       _KILL_THRESHOLD = 3

       def error_log_path() -> Path: return sekha_home() / "hook-errors.log"
       def marker_path() -> Path: return sekha_home() / "hook-disabled.marker"

       def read_event(stream) -> dict:
           data = stream.read()
           if not data: raise ValueError("empty stdin")
           return json.loads(data)

       def emit_block(reason: str, stdout, stderr) -> int:
           payload = {"hookSpecificOutput": {"hookEventName": "PreToolUse",
                                             "permissionDecision": "deny",
                                             "permissionDecisionReason": reason}}
           stdout.write(json.dumps(payload))
           stderr.write(reason + "\n")
           return 2

       def emit_warn(message: str, stdout) -> int:
           payload = {"hookSpecificOutput": {"hookEventName": "PreToolUse",
                                             "additionalContext": message}}
           stdout.write(json.dumps(payload))
           return 0

       def emit_allow(stdout) -> int:
           return 0

       def fail_open(exc: BaseException, stderr) -> int:
           path = error_log_path()
           path.parent.mkdir(parents=True, exist_ok=True)
           ts = datetime.now(timezone.utc).isoformat(timespec="seconds")
           with path.open("a", encoding="utf-8") as f:
               f.write(f"{ts} {type(exc).__name__}: {exc}\n")
               f.write(traceback.format_exc())
               f.write("\n")
           stderr.write(f"sekha hook error: {exc}\n")
           return 0

       def record_error(exc: BaseException) -> bool:
           """Append error to log. Return True if this trips the kill-switch threshold."""
           # fail_open already logged to the file. Check the tail for the threshold.
           path = error_log_path()
           if not path.exists(): return False
           # Read last ~20 lines; parse leading ISO timestamps.
           tail = path.read_text(encoding="utf-8").splitlines()[-40:]
           now = datetime.now(timezone.utc)
           recent = 0
           for line in tail:
               # Lines starting with timestamp have ISO format; others are traceback continuations.
               parts = line.split(" ", 1)
               if len(parts) < 2: continue
               try:
                   ts = datetime.fromisoformat(parts[0])
               except ValueError:
                   continue
               delta = (now - ts).total_seconds()
               if 0 <= delta <= _KILL_WINDOW_SECONDS:
                   recent += 1
           return recent >= _KILL_THRESHOLD

       def check_kill_switch() -> bool: return marker_path().exists()
       def create_marker() -> None:
           p = marker_path(); p.parent.mkdir(parents=True, exist_ok=True); p.touch()
       def clear_marker() -> None:
           p = marker_path()
           if p.exists(): p.unlink()
       ```

    2. GREEN — implement `src/sekha/hook.py`. Strict discipline: **only `sys` and `json` at module top**. Every other import lives inside `_run()`. Layout:

       ```python
       """PreToolUse hook entrypoint. Short-lived process invoked by Claude Code.

       Reads a PreToolUse JSON event from stdin, evaluates rules via sekha.rules,
       and emits a permissionDecision JSON to stdout. Fail-open on any internal
       error; kill-switch marker disables the hook after repeated failures.

       stdout is sacred — only the decision JSON goes there. sys.stdout is
       swapped to sys.stderr inside _run() before any heavyweight import so
       dependency log lines cannot corrupt the protocol stream.
       """
       # Requirement coverage:
       #   HOOK-01: `sekha hook run` entry (dispatched by sekha.cli)
       #   HOOK-02: Reads PreToolUse JSON from stdin
       #   HOOK-03: Emits deny JSON on block match
       #   HOOK-04: Emits additionalContext JSON on warn match
       #   HOOK-05: Belt-and-suspenders (stdout JSON + stderr + exit 2)
       #   HOOK-06: Fail-open with ~/.sekha/hook-errors.log + stderr warning
       #   HOOK-07: Kill switch after 3 errors in 10 min via marker file
       #   HOOK-09: Lazy imports — only sys, json at module top
       from __future__ import annotations

       import sys
       import json  # noqa: F401 — imported for the _run path; see note below

       # NOTE: json is needed by _run; we import it at module top deliberately
       # because (a) it's stdlib, (b) Python caches it anyway on first import,
       # and (c) CONTEXT.md permits sys + json at the top. All other imports
       # (sekha.rules, sekha.paths, sekha._hookutil, traceback, pathlib) live
       # inside _run() to hit the <30ms import budget.

       def _run(stdin, stdout, stderr) -> int:
           # Swap stdout before any non-stdlib import — guards against stray
           # logger output from sekha.rules / sekha.logutil reaching stdout.
           real_stdout = stdout
           # (tests inject a buffer; in main() this is sys.stdout)
           sys.stdout = stderr  # module-level sys.stdout is now stderr

           try:
               from sekha._hookutil import (
                   read_event, emit_block, emit_warn, emit_allow,
                   fail_open, record_error, check_kill_switch, create_marker,
               )
               # Kill switch — never evaluate rules if disabled.
               if check_kill_switch():
                   return emit_allow(real_stdout)

               try:
                   event = read_event(stdin)
                   from sekha.rules import load_rules, evaluate
                   from sekha.paths import category_dir

                   hook_event = event.get("hook_event_name", "PreToolUse")
                   tool_name = event.get("tool_name", "")
                   tool_input = event.get("tool_input", {}) or {}

                   rules = load_rules(category_dir("rules"), hook_event, tool_name)
                   winner = evaluate(rules, tool_input)
                   if winner is None:
                       return emit_allow(real_stdout)
                   if winner.severity == "block":
                       return emit_block(winner.message, real_stdout, stderr)
                   return emit_warn(winner.message, real_stdout)
               except Exception as exc:   # noqa: BLE001 — fail-open intentional
                   fail_open(exc, stderr)
                   if record_error(exc):
                       create_marker()
                   return 0
           finally:
               sys.stdout = real_stdout  # restore for process cleanliness

       """Sekha CLI router. `sekha <subcommand>` entry point.

       Subcommands live in their own modules and are lazy-imported so the CLI
       startup cost stays low. The existing pyproject.toml console-script
       `sekha = "sekha.cli:main"` dispatches here.

       Phase 4 subcommands: hook run | hook bench | hook enable | hook disable.
       Phase 6 will add: init, doctor, add-rule, list-rules, rule test, pause.
       """
       # Requirement coverage:
       #   HOOK-01: `sekha hook run` entry point registered via argparse + pyproject.
       from __future__ import annotations

       import argparse
       import sys


       def _build_parser() -> argparse.ArgumentParser:
           parser = argparse.ArgumentParser(prog="sekha", description="Sekha CLI")
           sub = parser.add_subparsers(dest="command", required=True)

           hook = sub.add_parser("hook", help="Hook operations")
           hook_sub = hook.add_subparsers(dest="hook_command", required=True)
           hook_sub.add_parser("run",     help="Run PreToolUse hook (stdin → decision JSON → stdout)")
           hook_sub.add_parser("bench",   help="Benchmark hook latency (100 runs, p50/p95/p99)")
           hook_sub.add_parser("enable",  help="Clear kill-switch marker; re-enable hook")
           hook_sub.add_parser("disable", help="Create kill-switch marker; short-circuit hook to allow")

           return parser


       def main(argv: list[str] | None = None) -> int:
           parser = _build_parser()
           args = parser.parse_args(argv)

           if args.command == "hook":
               if args.hook_command == "run":
                   from sekha.hook import main as hook_main
                   return hook_main()
               if args.hook_command == "bench":
                   try:
                       from sekha.hook import bench as hook_bench
                   except ImportError:
                       sys.stderr.write("sekha hook bench: not yet implemented (lands in plan 04-02)\n")
                       return 1
                   return hook_bench()
               if args.hook_command == "enable":
                   from sekha.hook import enable as hook_enable
                   return hook_enable()
               if args.hook_command == "disable":
                   from sekha.hook import disable as hook_disable
                   return hook_disable()

           parser.error(f"unknown command: {args.command}")
           return 2  # unreachable — parser.error raises SystemExit


       if __name__ == "__main__":
           raise SystemExit(main())
       ```
